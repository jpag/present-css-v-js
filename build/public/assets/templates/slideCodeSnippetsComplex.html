jQuery 

$(".block").each(function(t,el){
    setTimeout(function(){
        jqueryAnimate(el);
    }, (25 * t) );
});

jqueryAnimate : function(el) {
    var tp = 20,
        totalTime = 500,
        self = this;

    $(el).animate({
        top : tp,
        opacity : .5
    },{
        duration : totalTime/2,
        complete: function() {
            // go back:
            $(el).animate({
                top : 0,
                opacity : 1
            },{
                duration : totalTime/2,
                complete : function(){
                    self.jqueryAnimate(el);
                }
            });
        }
    });
}
